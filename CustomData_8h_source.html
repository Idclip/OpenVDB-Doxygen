132  template <typename TypedDataCacheT>
133  inline TypedDataCacheT*
134  getOrInsertData(const Name& name)
135  {
136  const auto iter = mData.find(name);
137  if (iter == mData.end()) {
138  Metadata::Ptr data(new TypedDataCacheT());
139  mData[name] = data;

152  template <typename TypedDataCacheT>
153  inline void
154  insertData(const Name& name,
155  const typename TypedDataCacheT::Ptr data)
156  {
157  if (hasData(name)) {
158  TypedDataCacheT* const dataToSet =
159  getOrInsertData<TypedDataCacheT>(name);
160  if (!dataToSet) {
161  OPENVDB_THROW(TypeError, "Custom data \"" + name + "\" already exists with a different type.");
162  }
163  dataToSet->value() = data->value();
164  }
165  else {
166  mData[name] = data->copy();
167  }
168  }
