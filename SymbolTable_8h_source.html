197  inline llvm::Value* find(const std::string& name, const size_t startIndex) const
198  {
199  // Find the lower bound start index and if necessary, decrement into
200  // the first block where the search will be started. Note that this
201  // is safe as the global block 0 will always exist
202 
203  auto it = mTables.lower_bound(startIndex);
204  if (it == mTables.end() || it->first != startIndex) --it;
205 
206  // reverse the iterator (which also make it point to the preceding
207  // value, hence the crement)
208 
209  assert(it != mTables.end());
210  MapType::const_reverse_iterator iter(++it);
211 
212  for (; iter != mTables.crend(); ++iter) {
213  llvm::Value* value = iter->second.get(name);
214  if (value) return value;
215  }
216 
217  return nullptr;
218  }
