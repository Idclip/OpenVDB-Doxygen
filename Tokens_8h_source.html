91 inline CoreType tokenFromTypeString(const std::string& type)
92 {
93  if (type[0] == 'v') {
94  if (type == "vec2i") return VEC2I;
95  if (type == "vec2f") return VEC2F;
96  if (type == "vec2d") return VEC2D;
97  if (type == "vec3i") return VEC3I;
98  if (type == "vec3f") return VEC3F;
99  if (type == "vec3d") return VEC3D;
100  if (type == "vec4i") return VEC4I;
101  if (type == "vec4f") return VEC4F;
102  if (type == "vec4d") return VEC4D;
103  }
104  else if (type[0] == 'm') {
105  if (type == "mat3f") return MAT3F;
106  if (type == "mat3d") return MAT3D;
107  if (type == "mat4f") return MAT4F;
108  if (type == "mat4d") return MAT4D;
109  }
110  else if (type[0] == 'q') {
111  if (type == "quatf") return QUATF;
112  if (type == "quatd") return QUATD;
113  }
114  else if (type == "bool") return BOOL;
115  else if (type == "char") return CHAR;
116  else if (type == "short") return SHORT;
117  else if (type == "int") return INT;
118  else if (type == "long") return LONG;
119  else if (type == "float") return FLOAT;
120  else if (type == "double") return DOUBLE;
121  else if (type == "string") return STRING;
122 
123  // also handle vdb types that have different type strings to our tokens
124  // @todo These should probably be separated out. The executables currently
125  // use this function to guarantee conversion
126  if (type[0] == 'v') {
127  if (type == "vec2s") return VEC2F;
128  if (type == "vec3s") return VEC3F;
129  if (type == "vec4s") return VEC4F;
130  }
131  else if (type[0] == 'm') {
132  if (type == "mat3s") return MAT3F;
133  if (type == "mat4s") return MAT4F;
134  }
135  else if (type == "quats") return QUATF;
136  else if (type[0] == 'i') {
137  if (type == "int16") return SHORT;
138  if (type == "int32") return INT;
139  if (type == "int64") return LONG;
140  }
141 
142  return UNKNOWN;
143 }

260 inline OperatorType operatorType(const OperatorToken token)
261 {
262  const size_t idx = static_cast<size_t>(token);
263  if (idx <= static_cast<size_t>(MODULO)) return ARITHMETIC;
264  if (idx <= static_cast<size_t>(NOT)) return LOGICAL;
265  if (idx <= static_cast<size_t>(LESSTHANOREQUAL)) return RELATIONAL;
266  if (idx <= static_cast<size_t>(BITNOT)) return BITWISE;
267  if (idx <= static_cast<size_t>(BITOREQUALS)) return ASSIGNMENT;
268  return UNKNOWN_OPERATOR;
269 }

271 inline OperatorToken operatorTokenFromName(const std::string& name)
272 {
273  if (name == "+") return PLUS;
274  if (name == "-") return MINUS;
275  if (name == "*") return MULTIPLY;
276  if (name == "/") return DIVIDE;
277  if (name == "%") return MODULO;
278  if (name == "&&") return AND;
279  if (name == "||") return OR;
280  if (name == "!") return NOT;
281  if (name == "==") return EQUALSEQUALS;
282  if (name == "!=") return NOTEQUALS;
283  if (name == ">") return MORETHAN;
284  if (name == "<") return LESSTHAN;
285  if (name == ">=") return MORETHANOREQUAL;
286  if (name == "<=") return LESSTHANOREQUAL;
287  if (name == "<<") return SHIFTLEFT;
288  if (name == ">>") return SHIFTRIGHT;
289  if (name == "&") return BITAND;
290  if (name == "|") return BITOR;
291  if (name == "^") return BITXOR;
292  if (name == "~") return BITNOT;
293  if (name == "=") return EQUALS;
294  if (name == "+=") return PLUSEQUALS;
295  if (name == "-=") return MINUSEQUALS;
296  if (name == "*=") return MULTIPLYEQUALS;
297  if (name == "/=") return DIVIDEEQUALS;
298  if (name == "%=") return MODULOEQUALS;
299  if (name == "<<=") return SHIFTLEFTEQUALS;
300  if (name == ">>=") return SHIFTRIGHTEQUALS;
301  if (name == "&=") return BITANDEQUALS;
302  if (name == "^=") return BITXOREQUALS;
303  if (name == "|=") return BITOREQUALS;
304  OPENVDB_THROW(AXSyntaxError, "Unsupported op \"" + name + "\"");
305 }
