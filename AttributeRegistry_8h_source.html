202 inline AttributeRegistry::Ptr AttributeRegistry::create(const ast::Tree& tree)
203 {
204  AttributeRegistry::Ptr registry(new AttributeRegistry());
205  std::vector<std::string> read, write, all;
206  ast::catalogueAttributeTokens(tree, &read, &write, &all);
207 
208  size_t idx = 0;
209  std::unordered_map<std::string, size_t> indexmap;
210 
211  auto dataBuilder =
212  [&](const std::vector<std::string>& attribs,
213  const bool readFlag,
214  const bool writeFlag)
215  {
216  std::string name, type;
217  for (const auto& attrib : attribs) {
218  ast::Attribute::nametypeFromToken(attrib, &name, &type);
219  const ast::tokens::CoreType typetoken =
220  ast::tokens::tokenFromTypeString(type);
221  registry->addData(name, typetoken, readFlag, writeFlag);
222  indexmap[attrib] = idx++;
223  }
224  };
225 
226  // insert all data
227 
228  dataBuilder(read, true, false);
229  dataBuilder(write, false, true);
230  dataBuilder(all, true, true);
231 
232  auto depBuilder = [&](const std::vector<std::string>& attribs) {
233 
234  std::string name, type;
235  for (const auto& attrib : attribs) {
236  ast::Attribute::nametypeFromToken(attrib, &name, &type);
237  const ast::tokens::CoreType typetoken =
238  ast::tokens::tokenFromTypeString(type);
239 
240  std::vector<std::string> deps;
241  ast::attributeDependencyTokens(tree, name, typetoken, deps);
242  if (deps.empty()) continue;
243 
244  assert(indexmap.find(attrib) != indexmap.cend());
245  const size_t index = indexmap.at(attrib);
246  AccessData& access = registry->mAccesses[index];
247  for (const std::string& dep : deps) {
248  assert(indexmap.find(dep) != indexmap.cend());
249  const size_t depindex = indexmap.at(dep);
250  access.mDependencies.emplace_back(&registry->mAccesses[depindex]);
251  }
252  }
253  };
254 
255  // initialize dependencies
256 
257  depBuilder(read);
258  depBuilder(write);
259  depBuilder(all);
260 
261  // Update usage from deps
262 
263  for (AccessData& access : registry->mAccesses) {
264  for (const AccessData& next : registry->mAccesses) {
265  // don't skip self depends as it may write to itself
266  // i.e. @a = @a + 1; should add a self usage
267  if (next.dependson(&access)) {
268  access.mUses.emplace_back(&next);
269  }
270  }
271  }
272 
273  return registry;
274 }

276 inline void AttributeRegistry::print(std::ostream& os) const
277 {
278  size_t idx = 0;
279  for (const auto& data : mAccesses) {
280  os << "Attribute: " << data.name() << ", type: " <<

284  os << " " << "Reads From : " << data.reads() << '\n';

286  os << std::noboolalpha;
287  os << " " << "Dependencies : " << data.mDependencies.size() << '\n';
288  for (const auto& dep : data.mDependencies) {
289  os << " " << "Attribute: " << dep->name() << " type: " <<
290  ast::tokens::typeStringFromToken(dep->type()) << '\n';
291  }
292  os << " " << "Usage : " << data.mUses.size() << '\n';
293  for (const auto& dep : data.mUses) {
294  os << " " << "Attribute: " << dep->name() << " type: " <<
295  ast::tokens::typeStringFromToken(dep->type()) << '\n';
296  }
297  os << '\n';
298  ++idx;
299  }
