147 inline llvm::Value*
148 insertStaticAlloca(llvm::IRBuilder<>& B,
149  llvm::Type* type,
150  llvm::Value* size = nullptr)
151 {
152  // Create the allocation at the start of the function block
153  llvm::Function* parent = B.GetInsertBlock()->getParent();
154  assert(parent && !parent->empty());
155  auto IP = B.saveIP();
156  llvm::BasicBlock& block = parent->front();
157  if (block.empty()) B.SetInsertPoint(&block);
158  else B.SetInsertPoint(&(block.front()));
159  llvm::Value* result = B.CreateAlloca(type, size);
160  B.restoreIP(IP);
161  return result;
162 }

334 inline BinaryFunction
335 llvmBinaryConversion(const llvm::Type* const type,
336  const ast::tokens::OperatorToken& token,
337  const std::string& twine = "")
338 {
339 
340 #define BIND_BINARY_OP(Function) \
341  [twine](llvm::IRBuilder<>& B, llvm::Value* L, llvm::Value* R) \
342  -> llvm::Value* { return B.Function(L, R, twine); }
343 
344  // NOTE: Binary % and / ops always take sign into account (CreateSDiv vs CreateUDiv, CreateSRem vs CreateURem).
345  // See http://stackoverflow.com/questions/5346160/llvm-irbuildercreateudiv-createsdiv-createexactudiv
346 
347  if (type->isFloatingPointTy()) {
348  const ast::tokens::OperatorType opType = ast::tokens::operatorType(token);
349  if (opType == ast::tokens::LOGICAL || opType == ast::tokens::BITWISE) {
350  OPENVDB_THROW(LLVMBinaryOperationError, "Unable to perform operation \""
351  + ast::tokens::operatorNameFromToken(token) + "\" on floating points values");
352  }
353 
354  if (token == ast::tokens::PLUS) return BIND_BINARY_OP(CreateFAdd);
355  else if (token == ast::tokens::MINUS) return BIND_BINARY_OP(CreateFSub);
356  else if (token == ast::tokens::MULTIPLY) return BIND_BINARY_OP(CreateFMul);
357  else if (token == ast::tokens::DIVIDE) return BIND_BINARY_OP(CreateFDiv);
358  else if (token == ast::tokens::MODULO) return BIND_BINARY_OP(CreateFRem);
359  else if (token == ast::tokens::EQUALSEQUALS) return BIND_BINARY_OP(CreateFCmpOEQ);
360  else if (token == ast::tokens::NOTEQUALS) return BIND_BINARY_OP(CreateFCmpONE);
361  else if (token == ast::tokens::MORETHAN) return BIND_BINARY_OP(CreateFCmpOGT);
362  else if (token == ast::tokens::LESSTHAN) return BIND_BINARY_OP(CreateFCmpOLT);
363  else if (token == ast::tokens::MORETHANOREQUAL) return BIND_BINARY_OP(CreateFCmpOGE);
364  else if (token == ast::tokens::LESSTHANOREQUAL) return BIND_BINARY_OP(CreateFCmpOLE);
365  OPENVDB_THROW(LLVMTokenError, "Unrecognised binary operator \"" +
366  ast::tokens::operatorNameFromToken(token) + "\"");
367  }
368  else if (type->isIntegerTy()) {
369  if (token == ast::tokens::PLUS) return BIND_BINARY_OP(CreateAdd); // No Unsigned/Signed Wrap
370  else if (token == ast::tokens::MINUS) return BIND_BINARY_OP(CreateSub); // No Unsigned/Signed Wrap
371  else if (token == ast::tokens::MULTIPLY) return BIND_BINARY_OP(CreateMul); // No Unsigned/Signed Wrap
372  else if (token == ast::tokens::DIVIDE) return BIND_BINARY_OP(CreateSDiv); // IsExact = false - when true, poison value if the reuslt is rounded
373  else if (token == ast::tokens::MODULO) return BIND_BINARY_OP(CreateSRem);
374  else if (token == ast::tokens::EQUALSEQUALS) return BIND_BINARY_OP(CreateICmpEQ);
375  else if (token == ast::tokens::NOTEQUALS) return BIND_BINARY_OP(CreateICmpNE);
376  else if (token == ast::tokens::MORETHAN) return BIND_BINARY_OP(CreateICmpSGT);
377  else if (token == ast::tokens::LESSTHAN) return BIND_BINARY_OP(CreateICmpSLT);
378  else if (token == ast::tokens::MORETHANOREQUAL) return BIND_BINARY_OP(CreateICmpSGE);
379  else if (token == ast::tokens::LESSTHANOREQUAL) return BIND_BINARY_OP(CreateICmpSLE);
380  else if (token == ast::tokens::AND) return BIND_BINARY_OP(CreateAnd);
381  else if (token == ast::tokens::OR) return BIND_BINARY_OP(CreateOr);
382  else if (token == ast::tokens::SHIFTLEFT) return BIND_BINARY_OP(CreateShl); // No Unsigned/Signed Wrap
383  else if (token == ast::tokens::SHIFTRIGHT) return BIND_BINARY_OP(CreateAShr); // IsExact = false - poison value if any of the bits shifted out are non-zero.
384  else if (token == ast::tokens::BITAND) return BIND_BINARY_OP(CreateAnd);
385  else if (token == ast::tokens::BITOR) return BIND_BINARY_OP(CreateOr);
386  else if (token == ast::tokens::BITXOR) return BIND_BINARY_OP(CreateXor);
387  OPENVDB_THROW(LLVMTokenError, "Unrecognised binary operator \"" +
388  ast::tokens::operatorNameFromToken(token) + "\"");
389  }
390 
391 #undef BIND_BINARY_OP
392 
393  std::cerr << "Attempted to generate a binary operator \""
394  << ast::tokens::operatorNameFromToken(token) << "\""
395  << "with type ";
396 
397  type->print(llvm::errs());
398 
399  OPENVDB_THROW(LLVMTypeError, "Invalid type for binary operation");
400 }

404 inline bool isValidCast(llvm::Type* from, llvm::Type* to)
405 {
406  assert(from && "llvm Type 'from' is null in isValidCast");
407  assert(to && "llvm Type 'to' is null in isValidCast");
408 
409  if ((from->isIntegerTy() || from->isFloatingPointTy()) &&
410  (to->isIntegerTy() || to->isFloatingPointTy())) {
411  return true;
412  }
413  if (from->isArrayTy() && to->isArrayTy()) {
414  llvm::ArrayType* af = llvm::cast<llvm::ArrayType>(from);
415  llvm::ArrayType* at = llvm::cast<llvm::ArrayType>(to);
416  if (af->getArrayNumElements() == at->getArrayNumElements()) {
417  return isValidCast(af->getArrayElementType(),
418  at->getArrayElementType());
419  }
420  }
421  return false;
422 }

457 inline llvm::Value*
458 arrayCast(llvm::Value* ptrToArray,
459  llvm::Type* targetElementType,
460  llvm::IRBuilder<>& builder)
461 {
462  assert(targetElementType && (targetElementType->isIntegerTy() ||
463  targetElementType->isFloatingPointTy()) &&
464  "Target element type is not a scalar type");
465  assert(ptrToArray && ptrToArray->getType()->isPointerTy() &&
466  "Input to arrayCast is not a pointer type.");
467 
468  llvm::Type* arrayType = ptrToArray->getType()->getContainedType(0);
469  assert(arrayType && llvm::isa<llvm::ArrayType>(arrayType));
470 
471  // getArrayElementType() calls getContainedType(0)
472  llvm::Type* sourceElementType = arrayType->getArrayElementType();
473  assert(sourceElementType && (sourceElementType->isIntegerTy() ||
474  sourceElementType->isFloatingPointTy()) &&
475  "Source element type is not a scalar type");
476 
477  if (sourceElementType == targetElementType) return ptrToArray;
478 
479  CastFunction llvmCastFunction = llvmArithmeticConversion(sourceElementType, targetElementType);
480 
481  const size_t elementSize = arrayType->getArrayNumElements();
482  llvm::Value* targetArray =
483  insertStaticAlloca(builder,
484  llvm::ArrayType::get(targetElementType, elementSize));
485 
486  for (size_t i = 0; i < elementSize; ++i) {
487  llvm::Value* target = builder.CreateConstGEP2_64(targetArray, 0, i);
488  llvm::Value* source = builder.CreateConstGEP2_64(ptrToArray, 0, i);
489  source = builder.CreateLoad(source);
490  source = llvmCastFunction(builder, source, targetElementType);
491  builder.CreateStore(source, target);
492  }
493 
494  return targetArray;
495 }

615  ast::tokens::operatorNameFromToken(token) + "\". " + os.str());
616  }

622  rhs = boolComparison(rhs, builder);
623  lhsType = lhs->getType();

628  lhsType = lhs->getType();
629  if (warnings) *warnings = std::string("Implicit cast from float to int.");
630  }

633  return llvmBinaryFunction(builder, lhs, rhs);
634 }
