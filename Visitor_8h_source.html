407  template <typename NodeT>
408  bool hierarchyVisits(NodeT& node)
409  {
410  if (this->derived().reverseHierarchyVisits()) {
411  if (auto base = node.NodeT::basetype()) {
412  if (!hierarchyVisits(*base)) return false;
413  }
414  if (!this->derived().visit(this->strip<ConstVisit>(node))) return false;
415  }
416  else {
417  if (!this->derived().visit(this->strip<ConstVisit>(node))) return false;
418  if (auto base = node.NodeT::basetype()) {
419  return hierarchyVisits(*base);
420  }
421  }
422  return true;
423  }

430  template <typename NodeT>
431  inline bool defaultTraversal(NodeType<NodeT>* node)
432  {
433  if (!node) return true;
434  const size_t children = node->children();
435 
436  if (this->derived().postOrderNodes()) {
437  if (this->derived().reverseChildVisits()) {
438  if (children != 0) {
439  for (int64_t i = static_cast<int64_t>(children - 1); i >= 0; --i) {
440  auto child = this->strip<ConstVisit>(*(node->child(i)));
441  if (!this->derived().traverse(child)) {
442  return false;
443  }
444  }
445  }
446  }
447  else {
448  for (size_t i = 0; i < children; ++i) {
449  auto child = this->strip<ConstVisit>(*(node->child(i)));
450  if (!this->derived().traverse(child)) {
451  return false;
452  }
453  }
454  }
455  if (this->derived().visitNodeHierarchies()) {
456  return this->hierarchyVisits(*node);
457  }
458  else {
459  return this->derived().visit(node);
460  }
461  }
462  else {
463  if (this->derived().visitNodeHierarchies()) {
464  if (!this->hierarchyVisits(*node)) return false;
465  }
466  else {
467  if (!this->derived().visit(node)) return false;
468  }
469  if (this->derived().reverseChildVisits()) {
470  if (children != 0) {
471  for (int64_t i = static_cast<int64_t>(children - 1); i >= 0; --i) {
472  auto child = this->strip<ConstVisit>(*(node->child(i)));
473  if (!this->derived().traverse(child)) {
474  return false;
475  }
476  }
477  }
478  }
479  else {
480  for (size_t i = 0; i < children; ++i) {
481  auto child = this->strip<ConstVisit>(*(node->child(i)));
482  if (!this->derived().traverse(child)) {
483  return false;
484  }
485  }
486  }
487  return true;
488  }

openvdb::v7_0::ax::ast::Visitor::reverseHierarchyVisits
bool reverseHierarchyVisits() const
Default behavior option. Reverses the traversal order of node hierarchies. If true, hierarchical visits start at the very top of their inheritance structure (always a Node AST node) and visit downwards until the lowest derived concrete node is reached. If false, hierarchical visits start at the lowest derived concrete node and visit upwards until the very top of their inheritance structure (always a Node AST node) is reached.
