102  inline GroupHandleT* getOrInsert(const std::string& name)
103  {
104  GroupHandleT* ptr = get(name);

107  static const size_t maxGroupsInArray =
108 #if (OPENVDB_LIBRARY_MAJOR_VERSION_NUMBER > 7 || \
109  (OPENVDB_LIBRARY_MAJOR_VERSION_NUMBER >= 7 && \
110  OPENVDB_LIBRARY_MINOR_VERSION_NUMBER >= 1))
111  points::AttributeSet::Descriptor::groupBits();
112 #else
113  // old removed method
114  points::point_group_internal::GroupInfo::groupBits();
115 #endif
116 
117  if (mArrays.empty() || mOffset == maxGroupsInArray) {
118  assert(mPointCount < static_cast<size_t>(std::numeric_limits<openvdb::Index>::max()));
119  mArrays.emplace_back(new GroupArrayT(static_cast<openvdb::Index>(mPointCount)));
120  mOffset = 0;
121  }
122 
123  GroupArrayT* array = mArrays.back().get();
124  assert(array);
125 
126  std::unique_ptr<GroupHandleT>& handle = mHandles[name];
127  handle.reset(new GroupHandleT(*array, mOffset++));
128  return handle.get();
129  }
