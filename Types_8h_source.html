85  static inline llvm::Type*
86  get(llvm::LLVMContext& C)
87  {
88  // @note bools always treated as i1 values as the constants
89  // true and false from the IRBuilder are i1
90  if (std::is_same<T, bool>::value) {
91  return llvm::Type::getInt1Ty(C);
92  }
93 
94 #if LLVM_VERSION_MAJOR > 6
95  return llvm::Type::getScalarTy<T>(C);
96 #else
97  int bits = sizeof(T) * CHAR_BIT;
98  if (std::is_integral<T>::value) {
99  return llvm::Type::getIntNTy(C, bits);
100  }
101  else if (std::is_floating_point<T>::value) {
102  switch (bits) {
103  case 32: return llvm::Type::getFloatTy(C);
104  case 64: return llvm::Type::getDoubleTy(C);
105  }
106  }
107  OPENVDB_THROW(LLVMTypeError, "LLVMType called with an unsupported type \"" +
108  std::string(typeNameAsString<T>()) + "\".");
109 #endif
110  }

117  static inline llvm::Constant*
118  get(llvm::LLVMContext& C, const T V)
119  {
120  llvm::Type* type = LLVMType<T>::get(C);
121  llvm::Constant* constant = nullptr;
122 
123  if (std::is_floating_point<T>::value) {
124  assert(llvm::ConstantFP::isValueValidForType(type,
125  llvm::APFloat(static_cast<typename std::conditional
126  <std::is_floating_point<T>::value, T, double>::type>(V))));
127  constant = llvm::ConstantFP::get(type, static_cast<double>(V));
128  }
129  else if (std::is_integral<T>::value) {
130  const constexpr bool isSigned = std::is_signed<T>::value;
131  assert((isSigned && llvm::ConstantInt::isValueValidForType(type, static_cast<int64_t>(V))) ||
132  (!isSigned && llvm::ConstantInt::isValueValidForType(type, static_cast<uint64_t>(V))));
133  constant = llvm::ConstantInt::get(type, static_cast<uint64_t>(V), isSigned);
134  }
135 
136  assert(constant);
137  return constant;
138  }

400 inline ast::tokens::CoreType
401 tokenFromLLVMType(const llvm::Type* type)
402 {
403  if (type->isPointerTy()) {
404  type = type->getPointerElementType();
405  }
406  if (type->isIntegerTy(1)) return ast::tokens::BOOL;
407  if (type->isIntegerTy(16)) return ast::tokens::SHORT;
408  if (type->isIntegerTy(32)) return ast::tokens::INT;
409  if (type->isIntegerTy(64)) return ast::tokens::LONG;
410  if (type->isFloatTy()) return ast::tokens::FLOAT;
411  if (type->isDoubleTy()) return ast::tokens::DOUBLE;
412  if (type->isArrayTy()) {
413  const ast::tokens::CoreType elementType =
414  tokenFromLLVMType(type->getArrayElementType());
415  const size_t size = type->getArrayNumElements();
416  if (size == 2) {
417  if (elementType == ast::tokens::INT) return ast::tokens::VEC2I;
418  if (elementType == ast::tokens::FLOAT) return ast::tokens::VEC2F;
419  if (elementType == ast::tokens::DOUBLE) return ast::tokens::VEC2D;
420  }
421  else if (size == 3) {
422  if (elementType == ast::tokens::INT) return ast::tokens::VEC3I;
423  if (elementType == ast::tokens::FLOAT) return ast::tokens::VEC3F;
424  if (elementType == ast::tokens::DOUBLE) return ast::tokens::VEC3D;
425  }
426  else if (size == 4) {
427  if (elementType == ast::tokens::INT) return ast::tokens::VEC4I;
428  if (elementType == ast::tokens::FLOAT) return ast::tokens::VEC4F;
429  if (elementType == ast::tokens::DOUBLE) return ast::tokens::VEC4D;
430  }
431  else if (size == 9) {
432  if (elementType == ast::tokens::FLOAT) return ast::tokens::MAT3F;
433  if (elementType == ast::tokens::DOUBLE) return ast::tokens::MAT3D;
434  }
435  else if (size == 16) {
436  if (elementType == ast::tokens::FLOAT) return ast::tokens::MAT4F;
437  if (elementType == ast::tokens::DOUBLE) return ast::tokens::MAT4D;
438  }
439  }
440  if (type == LLVMType<AXString>::get(type->getContext())) {
441  return ast::tokens::STRING;
442  }
443  return ast::tokens::UNKNOWN;
444 }
445 
446 }
